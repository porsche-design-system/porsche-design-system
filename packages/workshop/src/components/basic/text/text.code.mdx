# Text

## Text sizes and styles

### Size 1 - bold / regular

<Example noTheme={false} className="sg-example-text">
  {(theme) => {
    const setTheme = theme === "inverted" ? true : false
    return (
      <React.Fragment>
        <Text type="1-bold" as="h1" inverted={setTheme}>Lorem ipsum dolor sit amet</Text>
        <Text type="1-regular" as="h1" inverted={setTheme}>Lorem ipsum dolor sit amet</Text>
      </React.Fragment>
    )
  }}
</Example>
